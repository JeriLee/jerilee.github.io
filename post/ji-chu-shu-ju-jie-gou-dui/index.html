堆

原理

堆属于二叉树，属于完全二叉树。
堆特性 ： 以大顶堆为例，父节点比两个子节点的值都要大。如图，5号节点，它的v值比10号节点和11号节点都要大。
实现方式： 通常采用数组实现。

C#代码


  public class Heap<T> where T: IComparable<T> {
    private List<T> list = new List<T>();
    private int count = 0;

    public Heap() {
    }

    public void Add(T value) {
      if(list.Count > count) {
        list[count] = value;
      } else {
        list.Add(value);
      }

      CheckAdd(count);
      ++count;
    }

    private void CheckAdd(int index) {
      if(index > 0) {
        int father = (index - 1 ) >> 1;
        if (list[index].CompareTo(list[father]) < 0) {
          T temp = list[index];
          list[index] = list[father];
          list[father] = temp;
          CheckAdd(father);
        }
      }
    }

    public T Top() {
      if(count > 0) {
        return list[0];
      }
      return default;
    }

    public int Count => count;
    public bool IsEmpty => count == 0;

    public bool Pop() {
      if(count > 0) {
        --count;
        if (count > 0) {
          T temp = list[0];
          list[0] = list[count];
          list[count] = temp;
          CheckPop(0);
        }
        return true;
      }
      return false;
    }

    private void CheckPop(int index) {
      int leftSon = index << 1 | 1;
      int rightSon = leftSon + 1;
      if (leftSon < Count) {
        int upSon = rightSon < Count && list[rightSon].CompareTo(list[leftSon]) < 0 ? rightSon : leftSon;
        if (list[upSon].CompareTo(list[index]) < 0) {
          T temp = list[index];
          list[index] = list[upSon];
          list[upSon] = temp;
          CheckPop(upSon);
        }
      }
    }
  }
