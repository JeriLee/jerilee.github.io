字典树 Trie

字典树应用：适合排除敏感词，适合模糊查询
字典树是一颗多叉树，一般采用一个字节大小作为字典树一个节点，即为256叉树。

逻辑实现

例如，现在有敏感词如下：
abc
abd
acd
bda
bef
orz
字典树结构如图，从根节点到每一个叶节点的路径为每个敏感词，比如root-o-r-z这条路径组成"orz"这个字符串

每次查询目标字符串时，从root开始向下匹配字符，时间复杂度O(deep)，deep为字典树深度，空间复杂度无法估算。

字典树查询字符串
  // Author : JeriLee
  // Time : 2020-7-18-11:01
  public class Trie {
    public Trie() {
    }

    private List<int[]> list_ = new List<int[]>();
    private List<bool> vis_ = new List<bool>();

    public void Init(IEnumerable<string> strs) {
      if(strs != null) {
        AddNode();
        foreach (var str in strs) {
          AddModelString(str);
        }
      }

      for(int index = 0; index < list_.Count; ++index) {
        var dig = list_[index];
        Console.WriteLine($"Node {index} has {vis_[index]}");
        foreach (var item in dig) {
          if(item > 0) {
            Console.WriteLine($" next{item}");
          }
        }
      }
    }

    private void AddModelString(string str) {
      if(str != null && str.Length > 0) {
        AddLastString(0, str, 0);
      }
    }

    private void AddLastString(int cnt, string str, int index) {
      if(index == str.Length) {
        vis_[cnt] = true;
      } else {
        char c = str[index];
        if(list_[cnt][c] == 0) {
          list_[cnt][c] = list_.Count;
          AddNode();
        }
        AddLastString(list_[cnt][c], str, index + 1);
      }
    }

    private void AddNode() {
      list_.Add(new int[1 << 8]);
      vis_.Add(default);
    }

    public int FindString(string str) {
      return FindStringByBord(str, 0, str.Length);
    }

    public int FindString(string str, int startIndex , int len) {
      int endIndex = startIndex + len;
      return FindStringByBord(str, startIndex, Math.Min(endIndex, str.Length));
    }

    private int FindStringByBord(string str, int index, int endIndex, int cnt = 0) {
      if (index == endIndex) {
        return index;
      }
      char c = str[index];
      if (list_[cnt][c] > 0) {
        return FindStringByBord(str, index + 1, endIndex, list_[cnt][c]);
      }
      return 0;
    }
  }
